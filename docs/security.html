    async def read_body(req: dict, *, max_size: int = MAX_BODY) -> bytes:    """Read the request body with caching and size enforcement.    Safe to call multiple times — the result is cached on first read.    Raises BodyTooLarge if the body exceeds max_size.    Drop-in replacement for the original ``body()`` function.    """    # Return cached result    if "_body_cache" in req:        return req["_body_cache"]    receive = req["internal_receive"]    chunks: list[bytes] = []    total = 0    while True:        msg = await receive()        chunk = msg.get("body", b"")        total += len(chunk)        if total > max_size:            req["_body_cache"] = b""  # prevent re-reads            raise BodyTooLarge(max_size)        chunks.append(chunk)        if not msg.get("more_body"):            break    result = b"".join(chunks)    req["_body_cache"] = result    return result
